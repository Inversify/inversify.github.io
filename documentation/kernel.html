    Inversify is a lightweight pico container for TypeScript and JavaScript apps. A pico container uses a class constructor to identify and inject its dependencies. For this to work, the class needs to declare a constructor that includes everything it needs injected. In order to resolve a depencency, the pico container needs to be told which implementation type (classes) to associate with each service type (interfaces).

    Examines if a constructor has any dependencies. If so, it will resolve and inject them

        Kernel.prototype._injectDependencies = function (func) {
        var args = this._getConstructorArguments(func);
        if (args.length === 0) {
            return new func();
        }
        else {
            var injections = [], implementation = null;
            for (var i = 0; i < args.length; i++) {
                var service = args[i];
                implementation = this.resolve(service); // TODO: remove any?
                injections.push(implementation);
            }
            return this._construct(func, injections);
        }
    };

        Kernel.prototype._construct = function (constr, args) {
        function F() {
            constr.apply(this, args);

        F.prototype = constr.prototype;
